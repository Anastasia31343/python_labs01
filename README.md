def json_to_csv(json_path: str, csv_path: str) -> None:

import csv
from pathlib import Path
from openpyxl import Workbook
from openpyxl.utils import get_column_letter

def csv_to_xlsx(csv_path: str, xlsx_path: str) -> None:
    if Path(csv_path).is_absolute() or Path(xlsx_path).is_absolute():
        raise ValueError("–ø—É—Ç–∏ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–º–∏")
    if not csv_path.endswith('.csv') or not xlsx_path.endswith('.xlsx'):
        raise ValueError("–Ω–µ–≤–µ—Ä–Ω—ã–µ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è —Ñ–∞–π–ª–æ–≤")
    if not Path(csv_path).exists():
        raise FileNotFoundError(f"—Ñ–∞–π–ª –Ω–µ –Ω–∞–π–¥–µ–Ω: {csv_path}")
    
    with open(csv_path, "r", encoding="utf-8") as f:
        rows = list(csv.reader(f))
    
    if not rows or not any(rows[0]):
        raise ValueError("–ø—É—Å—Ç–æ–π CSV –∏–ª–∏ –Ω–µ—Ç –∑–∞–≥–æ–ª–æ–≤–∫–∞")
    
    wb = Workbook()
    ws = wb.active
    ws.title = "Sheet1"
    
    max_lengths = []
    for row in rows:
        ws.append(row)
        for i, value in enumerate(row):
            if i >= len(max_lengths):
                max_lengths.append(0)
            max_lengths[i] = max(max_lengths[i], len(str(value or "")))
    
    for i, length in enumerate(max_lengths, 1):
        ws.column_dimensions[get_column_letter(i)].width = max(length + 2, 8)
    
    Path(xlsx_path).parent.mkdir(parents=True, exist_ok=True)
    wb.save(xlsx_path)

if __name__ == "__main__":
    Path("data/out").mkdir(parents=True, exist_ok=True)
    csv_to_xlsx("data/samples/people.csv", "data/out/people.xlsx")

def normalize(text):
    text = text.lower()

    return text.strip()

def tokenize(text):
    return re.findall(r'\b[a-z–∞-—è—ë0-9]+\b', text, re.IGNORECASE)

def count_freq(words):
    freq = {}
    for word in words:
        freq[word] = freq.get(word, 0) + 1
    return freq

def top_n(freq, n=5):
    sorted_items = sorted(freq.items(), key=lambda x: (-x[1], x[0]))
    return sorted_items[:n]

from typing import Any, Optional #Any ‚Äî –ª—é–±–æ–π —Ç–∏–ø –¥–∞–Ω–Ω—ã—Ö, Optional ‚Äî –º–æ–∂–µ—Ç –±—ã—Ç—å —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ —Ç–∏–ø–∞ –∏–ª–∏ None

class Node: #Node –¥–ª—è —É–∑–ª–∞ –æ–¥–Ω–æ—Å–≤—è–∑–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞
    def __init__(self, value: Any, next_node: Optional['Node'] = None): #'Node' ‚Äî —Ñ–æ—Ä–≤–∞—Ä–¥-–¥–µ–∫–ª–∞—Ä–∞—Ü–∏—è
        self.value = value #—Å–æ—Ö—Ä–∞–Ω—è–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –≤ –∞—Ç—Ä–∏–±—É—Ç–µ —É–∑–ª–∞
        self.next = next_node #—Å–æ—Ö—Ä–∞–Ω—è–µ—Ç —Å—Å—ã–ª–∫—É –Ω–∞ —Å–ª–µ–¥—É—é—â–∏–π —É–∑–µ–ª
    def __str__(self) -> str:
        return f"[{self.value}]" #–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å—Ç—Ä–æ–∫—É –≤ —Ñ–æ—Ä–º–∞—Ç–µ [–∑–Ω–∞—á–µ–Ω–∏–µ]

class SinglyLinkedList: #–°–æ–∑–¥–∞—ë—Ç –∫–ª–∞—Å—Å –¥–ª—è –æ–¥–Ω–æ—Å–≤—è–∑–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞
    def __init__(self):
        self.head: Optional[Node] = None #—Å—Å—ã–ª–∫–∞ –Ω–∞ –ø–µ—Ä–≤—ã–π —É–∑–µ–ª —Å–ø–∏—Å–∫–∞
        self.tail: Optional[Node] = None #—Å—Å—ã–ª–∫–∞ –Ω–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–π —É–∑–µ–ª —Å–ø–∏—Å–∫–∞
        self._size: int = 0 #—Å—á—ë—Ç—á–∏–∫ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤
    def append(self, value: Any) -> None:
        new_node = Node(value)
        if self.head is None:
            self.head = self.tail = new_node #–µ—Å–ª–∏ —Å–ø–∏—Å–æ–∫ –ø—É—Å—Ç, –Ω–æ–≤—ã–π —É–∑–µ–ª —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –∏ –≥–æ–ª–æ–≤–æ–π, –∏ —Ö–≤–æ—Å—Ç–æ–º
        else:
            self.tail.next = new_node #—Ç–µ–∫—É—â–∏–π —Ö–≤–æ—Å—Ç–æ–≤–æ–π —É–∑–µ–ª —Å—Å—ã–ª–∞–µ—Ç—Å—è –Ω–∞ –Ω–æ–≤—ã–π —É–∑–µ–ª
            self.tail = new_node #–Ω–æ–≤—ã–π —É–∑–µ–ª —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –Ω–æ–≤—ã–º —Ö–≤–æ—Å—Ç–æ–º
        self._size += 1
    def __str__(self) -> str:
        nodes = []
        current = self.head #–Ω–∞—á–∏–Ω–∞–µ–º —Å –≥–æ–ª–æ–≤—ã —Å–ø–∏—Å–∫–∞
        while current is not None:
            nodes.append(str(current)) #–¥–æ–±–∞–≤–ª—è–µ–º —Å—Ç—Ä–æ–∫–æ–≤–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —Ç–µ–∫—É—â–µ–≥–æ —É–∑–ª–∞
            current = current.next
        nodes.append("None") #–¥–æ–±–∞–≤–ª—è–µ–º "None" –≤ –∫–æ–Ω–µ—Ü
        return " -> ".join(nodes) 
    def __repr__(self) -> str: #–ú–µ—Ç–æ–¥ repr –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏ –∏ –ø–æ–Ω—è—Ç–Ω–æ–≥–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏—è –æ–±—ä–µ–∫—Ç–∞
        values = []
        current = self.head
        while current is not None:
            values.append(current.value)
            current = current.next
        return f"SinglyLinkedList({values})"
    def __len__(self) -> int: 
        return self._size #–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ —Å—á—ë—Ç—á–∏–∫–∞ _size
